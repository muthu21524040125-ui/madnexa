Recommended Firestore Security Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Orders: authenticated users can create, admin can do all
    match /orders/{id} {
      allow create: if request.auth != null;
      allow read, update, delete: if request.auth != null
        && request.auth.token.email == "YOUR_ADMIN_EMAIL";
    }

    // Messages: anyone can create (contact form), admin can manage
    match /messages/{id} {
      allow create: if true;
      allow read, update, delete: if request.auth != null
        && request.auth.token.email == "YOUR_ADMIN_EMAIL";
    }

    // Users: user can read/write own doc, admin can read all
    match /users/{uid} {
      allow read, write: if request.auth != null
        && (request.auth.uid == uid
           || request.auth.token.email == "YOUR_ADMIN_EMAIL");
    }
  }
}
